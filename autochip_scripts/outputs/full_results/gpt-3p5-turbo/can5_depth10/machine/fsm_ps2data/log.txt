                    next_state = 3'b000;
            end
            default: next_state = 3'b000;
        endcase
    end

    always @(posedge clk) begin
        if (reset) begin
            out_bytes <= 24'b0;
            done <= 0;
        end else if (state == 3'b011) begin
            out_bytes <= {out_bytes[15:0], in};
            done <= 1;
        end else begin
            out_bytes <= {out_bytes[15:0], in};
            done <= 0;
        end
    end

endmodule
