                        state <= 3'b001; // Go to B
                    else
                        state <= 3'b000; // Stay in A
                end

                3'b001: begin // B
                    if (w)
                        state <= 3'b010; // Go to C
                    else
                        state <= 3'b011; // Go to D
                end

                3'b010: begin // C
                    if (w)
                        state <= 3'b111; // Go to F
                    else
                        state <= 3'b110; // Go to E
                end

                3'b011: begin // D
                    if (w)
                        state <= 3'b000; // Go to A
                    else
                        state <= 3'b100; // Go to E
                end

                3'b100: begin // E
                    if (w)
                        state <= 3'b011; // Go to D
                    else
                        state <= 3'b100; // Stay in E
                end

                3'b101: begin // F
                    if (w)
                        state <= 3'b011; // Go to D
                    else
                        state <= 3'b010; // Go to C
                end

                default: state <= 3'b000; // Default state is A
            endcase
        end
    end

    always @(posedge clk) begin
        case (state)
            3'b000, 3'b011: z <= 0; // A, D
            3'b100, 3'b110: z <= 1; // E, C
            default: z <= 0;
        endcase
    end

endmodule
