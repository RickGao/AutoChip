            state <= WALK_RIGHT;

            state <= IDLE;
          end else if (dig) begin

          end
        end

            state <= IDLE;

            state <= IDLE;
          end else if (!ground) begin
            state <= IDLE;
          end else if (dig) begin

            state <= IDLE;

        aaah <= 1'b0;
        digging <= 1'b0;

  end

          end else if (ground && dig && prev_state == IDLE) begin
            walk_left <= 1'b0;
            walk_right <= 1'b0;
            aaah <= 1'b0;
            digging <= 1'b1;
            state <= IDLE;
          end else if (!ground && !dig && prev_state == WALK_LEFT) begin
            walk_left <= 1'b1;
            walk_right <= 1'b0;
            aaah <= 1'b0;
            digging <= 1'b0;
            state <= WALK_LEFT;
          end else if (!ground && !dig && prev_state == WALK_RIGHT) begin
            walk_left <= 1'b0;
            walk_right <= 1'b1;
            aaah <= 1'b0;
            digging <= 1'b0;
            state <= WALK_RIGHT;
          end
        end
        
        WALK_LEFT: begin
          if (bump_right) begin
            state <= IDLE;
          end else if (!ground && prev_state != WALK_LEFT) begin
            state <= prev_state;
          end else if (!ground && prev_state == WALK_LEFT) begin
            aaah <= 1'b1;
            state <= IDLE;
          end else if (ground && dig && prev_state != WALK_LEFT) begin
            state <= IDLE;
          end else if (ground && dig && prev_state == WALK_LEFT) begin
            walk_left <= 1'b0;
            walk_right <= 1'b0;
            aaah <= 1'b0;
            digging <= 1'b1;
            state <= IDLE;
          end
        end
        
        WALK_RIGHT: begin
          if (bump_left) begin
            state <= IDLE;
          end else if (!ground && prev_state != WALK_RIGHT) begin
            state <= prev_state;
          end else if (!ground && prev_state == WALK_RIGHT) begin
            aaah <= 1'b1;
            state <= IDLE;
          end else if (ground && dig && prev_state != WALK_RIGHT) begin
            state <= IDLE;
          end else if (ground && dig && prev_state == WALK_RIGHT) begin
            walk_left <= 1'b0;
            walk_right <= 1'b0;
            aaah <= 1'b0;
            digging <= 1'b1;
            state <= IDLE;
          end
        end
      endcase
    end
  end
  
  always @(negedge clk or posedge areset) begin
    if (areset) begin
      prev_state <= IDLE;
    end else begin
      prev_state <= state;
    end
  end
  
endmodule
