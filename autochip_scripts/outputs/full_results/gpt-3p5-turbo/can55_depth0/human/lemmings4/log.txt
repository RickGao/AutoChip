    localparam DIGGING = 3'b100;
    localparam SPLATTER = 3'b101;

            state <= WALK_LEFT;
        end else begin
            state <= next_state;
        end
    end

        next_state = state;
        walk_left = 0;
        walk_right = 0;
        aaah = 0;
        digging = 0;
        
        case (state)
            WALK_LEFT: begin
                if (bump_right) begin
                    next_state = WALK_RIGHT;
                end else if (ground == 0) begin
                    next_state = FALLING;
                end else if (dig) begin
                    next_state = DIGGING;
                end
                walk_left = 1;
            end
            WALK_RIGHT: begin
                if (bump_left) begin
