  input w,
  output reg Y2);

  // Flip-flop declarations
  reg [3:1] y_reg;
  reg [3:1] y_next;
  
  // Next-state logic
  always @(y or w) begin
    case (y)
      3'b000:
        y_next = (w) ? 3'b000 : 3'b001;
      3'b001:
        y_next = (w) ? 3'b010 : 3'b011;

      default:
        y_next = 3'b000;
    endcase
  end
  
  // Reg update
  always @(posedge w) begin
    y_reg <= y_next;
  end
