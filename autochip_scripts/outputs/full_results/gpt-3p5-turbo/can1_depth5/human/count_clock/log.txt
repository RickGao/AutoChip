                            min_ones <= 0;
                            min_tens <= min_tens + 1;

                            if (min_tens == 6) begin
                                min_tens <= 0;
                                hour_ones <= hour_ones + 1;

                                if (hour_ones == 10 && hour_tens != 1) begin
                                    hour_ones <= 0;
                                    hour_tens <= hour_tens + 1;
                                end

                                if (hour_tens == 1 && hour_ones == 3) begin
                                    hour_tens <= 0;

                                    if (pm) begin
                                        pm <= 0;
                                    end else begin
                                        pm <= 1;
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
    end

    always @(posedge clk or posedge reset) begin
        if (reset) begin
            hh <= 8'b00010010;
            mm <= 8'b00000000;
            ss <= 8'b00000000;
        end else if (ena && (current_sec == 2'b00)) begin
            hh <= {4'b0000, hour_tens, hour_ones};
            mm <= {4'b0000, min_tens, min_ones};
            ss <= {4'b0000, sec_tens, sec_ones};
        end
    end

endmodule

module top_module_tb();

    reg clk, reset, ena;
    wire [7:0] hh, mm, ss;
    wire pm;

    top_module dut (
        .clk(clk),
        .reset(reset),
        .ena(ena),
        .hh(hh),
        .mm(mm),
        .ss(ss),
        .pm(pm)
    );

    initial begin
        clk = 0;
        reset = 1;
        ena = 0;

        // Reset
        #5 reset = 0;

        // Enable
        #10 ena = 1;

        // Clock edge
        #5 ena = 0;
        #5 ena = 1;

        // Wait for 12 hours
        repeat(432_000) begin
            #1 clk = ~clk;
        end

        // Stop Simulation
        $finish;
    end

endmodule
