  input reset,
  output out

  reg [1:0] state;
  always @(posedge clk or posedge reset) begin
    if (reset)
      state <= 2'b00; // Reset to state A
    else begin
      case (state)
        2'b00: begin // State A
          if (in)
            state <= 2'b01; // Change to state B
        end

        2'b01: begin // State B
          if (in)

          else
            state <= 2'b10; // Change to state C
        end

          else
            state <= 2'b10; // Change to state C
        end
      endcase
    end
  end

  assign out = (state == 2'b11); // Output is 1 when in state D

endmodule
