          end
          state <= 3;
        end

            state <= 4;

            // Detected start bit for next byte

            state <= 1;

        end

          if (in == 0) begin
            // Detected start bit for next byte
            start_bit_detected <= 1;
            state <= 1;
          end
        end
      endcase
    end
  end
