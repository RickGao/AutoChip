    assign walk_right = (state == 2'b01);
    assign aaah = (state == 2'b10);

    // Define the state transitions and outputs

                        state <= 2'b01; // Switch to walk_right state

                    end else if (!ground) begin
                        state <= 2'b10; // Transition to aaah state
