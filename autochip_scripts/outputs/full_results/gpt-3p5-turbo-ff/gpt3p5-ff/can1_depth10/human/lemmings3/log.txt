                        aaah <= 1'b1;
                    end else if (dig && ground) begin
                        state <= 2'b10; // Stay in the walking right state
                        walk_left <= 1'b0;
                        walk_right <= 1'b1;
                        digging <= 1'b1;
                    end
                end
                2'b01: begin // Walking left state
                    if (bump_left && !bump_right) begin
                        state <= 2'b00; // Switch to walking right state
                        walk_left <= 1'b0;
                        walk_right <= 1'b1;
                    end else if (bump_right && !bump_left) begin
                        state <= 2'b01; // Stay in the walking left state
                        walk_left <= 1'b1;
                        walk_right <= 1'b0;
                    end else if (bump_left && bump_right) begin
                        state <= 2'b00; // Switch to walking right state
                        walk_left <= 1'b0;
                        walk_right <= 1'b1;
                    end else if (!ground) begin
                        state <= 2'b10; // Switch to falling state
                        walk_left <= 1'b0;
                        walk_right <= 1'b0;
                        aaah <= 1'b1;
                    end else if (dig && ground) begin
                        state <= 2'b10; // Stay in the walking left state
                        walk_left <= 1'b1;
                        walk_right <= 1'b0;
                        digging <= 1'b1;
                    end
                end
                2'b10: begin // Falling state
                    if (ground) begin
                        if (previous_ground) begin
                            state <= 2'b00; // Stay in the walking right state
                            walk_left <= 1'b0;
                            walk_right <= 1'b1;
                            aaah <= 1'b0;
                        end else begin
                            state <= 2'b01; // Stay in the walking left state
                            walk_left <= 1'b1;
                            walk_right <= 1'b0;
                            aaah <= 1'b0;
                        end
                        digging <= 1'b0;
                    end
                end
                default: begin // Default state (should never happen)
                    state <= 2'b10; // Start in walking left state
                    walk_left <= 1'b1;
                    walk_right <= 1'b0;
                    aaah <= 1'b0;
                    digging <= 1'b0;
                end
            endcase
        end
    end
endmodule
