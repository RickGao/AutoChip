                        next_q[i*16 + j] = q[i*16 + j];
                    end else if (num_neighbors == 3) begin
                        next_q[i*16 + j] = 1;
                    end else if (num_neighbors >= 4) begin
                        next_q[i*16 + j] = 0;
                    end
                end
            end
            
            q <= next_q;
        end
    end
endmodule
