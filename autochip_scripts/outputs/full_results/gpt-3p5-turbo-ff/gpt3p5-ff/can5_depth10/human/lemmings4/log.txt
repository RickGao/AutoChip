                        state <= 5'b00100; // Digging
                    end
                end

                        state <= 5'b00001; // Walk left
                    end
                    else if (!ground) begin
                        state <= 5'b00011; // Falling
                    end
                    else if (dig) begin
                        state <= 5'b00100; // Digging
                    end
                end

                // Falling state
                5'b00011: begin
                    if (ground) begin
                        state <= 5'b00001; // Walk left
                    end
                    else if (state == 5'b10000 && !ground) begin
                        state <= 5'b10000; // Splatter
                    end

                        state <= state + 1;
                    end
                end

                // Digging state
                5'b00100: begin
                    if (ground) begin
                        state <= 5'b00001; // Walk left
                    end
                    else if (state == 5'b10000 && !ground) begin
                        state <= 5'b10000; // Splatter
                    end
                    else begin
                        state <= state + 1;
                    end
                end

                // Splatter state
                5'b10000: begin
                    state <= 5'b10000; // Stay in splatter state
                end

            endcase

    end
