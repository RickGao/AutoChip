        if (reset)
            state <= 1'b1;
        else begin
            case(state)
                1'b0: begin
                    if (in)
                        state <= 1'b0;
                    else
                        state <= 1'b1;
                end
                1'b1: begin
                    if (in)
                        state <= 1'b1;

    end

    always @(posedge clk) begin
        case(state)
            1'b0: out <= 1'b0;
            1'b1: out <= 1'b1;

    end
endmodule
