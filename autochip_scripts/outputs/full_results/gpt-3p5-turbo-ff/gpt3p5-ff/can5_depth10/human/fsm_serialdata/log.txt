            3'b011: begin // Receiving start bit
                next_state = 3'b010;
            end
            3'b100: begin // Waiting for stop bit
                next_state = (stop_bit) ? 3'b000 : 3'b100;
            end
            default: next_state = 3'b000;
        endcase
    end
endmodule
