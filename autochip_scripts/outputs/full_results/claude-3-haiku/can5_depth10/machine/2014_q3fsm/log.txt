            end
            C: begin
                if (w) state <= S11;
                else state <= S10;
                z <= 1'b0;
            end
            S10: begin
                if (w) state <= S21;
                else state <= S20;
                z <= 1'b0;
            end
            S11: begin
                if (w) state <= S22;
                else state <= S21;
                z <= 1'b0;
            end
            S20: begin
                state <= B;
                z <= 1'b0;
            end
            S21: begin
                if (w) state <= C;
                else state <= B;
                z <= 1'b0;
            end
            S22: begin
                if (w) state <= B;
                else state <= C;
                z <= 1'b0;
            end
            default: begin
                state <= A;
                z <= 1'b0;
            end
        endcase
    end
end
