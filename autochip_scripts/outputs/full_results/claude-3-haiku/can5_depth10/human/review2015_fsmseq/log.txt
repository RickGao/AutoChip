    case (state)
        IDLE: begin
            start_shifting = 1'b0;
            if (data == 1'b1)
                next_state = S1;
            else
                next_state = IDLE;
        end
        S1: begin
            start_shifting = 1'b0;
            if (data == 1'b1)
                next_state = S2;
            else
                next_state = IDLE;
        end
        S2: begin
            start_shifting = 1'b0;
            if (data == 1'b0)
                next_state = S3;
            else
                next_state = IDLE;
        end
        S3: begin
            start_shifting = 1'b0;
            if (data == 1'b1)
                next_state = S4;
            else
                next_state = IDLE;
        end
        S4: begin
            start_shifting = 1'b1;
            next_state = FOUND;
        end
        FOUND: begin
            start_shifting = 1'b1;
            next_state = FOUND;
        end
        default: begin
            start_shifting = 1'b0;
            next_state = IDLE;
        end
    endcase
end

endmodule
