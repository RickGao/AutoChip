                 FIVE_1S = 3'b001,
                 SEVEN_1S = 3'b011,
                 EIGHT_1S = 3'b111;

reg [2:0] state, next_state;

    if (reset)
        state <= IDLE;
    else
        state <= next_state;

    disc = 0;
    flag = 0;
    err = 0;

        IDLE: begin
            if (in == 1'b1)
                next_state = FIVE_1S;
        end
        FIVE_1S: begin
            if (in == 1'b0)
                next_state = IDLE;
            else if (in == 1'b1)
                next_state = SEVEN_1S;
        end
        SEVEN_1S: begin
            if (in == 1'b0) begin
                next_state = IDLE;
                disc = 1;
            end
            else if (in == 1'b1)
                next_state = EIGHT_1S;
        end
        EIGHT_1S: begin
            if (in == 1'b0) begin
                next_state = IDLE;
                flag = 1;
            end
            else if (in == 1'b1)
                next_state = EIGHT_1S;
        end
    endcase

    if (state == EIGHT_1S && in == 1'b1)
        err = 1;
end
