localparam STATE_CONVERT = 2'b01;
localparam STATE_DONE = 2'b11;

// State register
reg [1:0] state;

        // Reset the state machine

                // Initialize the input number register and the state

                state <= STATE_CONVERT;

            STATE_CONVERT: begin
                // Shift the input bit into the number register
                num <= {num[6:0], x};

                // Output the least-significant bit of the 2's complement
                z <= complement[0];
                
                // Check if all bits have been processed
                if (num[7]) begin
                    state <= STATE_DONE;
                end
            end
            STATE_DONE: begin
                // Hold the output and wait for the next reset
                z <= 1'b0;
            end
            default: begin
                // Invalid state, go to the initial state
                state <= STATE_INIT;
            end
        endcase
